19  package com.github.essobedo.gitlabvm;
20  
21  import com.github.essobedo.appma.exception.ApplicationException;
22  import com.goebl.david.Response;
23  import com.goebl.david.Webb;
24  import com.goebl.david.WebbException;
25  import java.io.InputStream;
26  import java.io.UnsupportedEncodingException;
27  import java.net.URLEncoder;
28  import java.util.Comparator;
29  import java.util.SortedSet;
30  import java.util.TreeSet;
31  import org.json.JSONArray;
32  import org.json.JSONException;
33  import org.json.JSONObject;
34  
35  /**
36   * The class allowing to access to gitlab using the Gitlab API.
37   *
38   * @author Nicolas Filotto (nicolas.filotto@gmail.com)
39   * @version $Id$
40   * @since 1.0
41   */
42  final class Repository {
43  
44      /**
45       * The HTTP code in case of an OK.
46       */
47      private static final int OK_CODE = 200;
48      /**
49       * The min value of the HTTP codes from which we consider the response as a success.
50       */
51      private static final int MIN_SUCCESS_CODE = 400;
52      /**
53       * The encoding used to URL encode the parameters.
54       */
55      private static final String ENCODING = "UTF-8";
56      /**
57       * The private token.
58       */
59      private String token;
60      /**
61       * The {@link Webb} instance allowing to access to gitlab thanks to the http/https protocol.
62       */
63      private final Webb webb;
64      /**
65       * The configuration to use to access to gitlab.
66       */
67      private final ConnectionConfiguration configuration;
68  
69      /**
70       * Constructs a {@code Repository} with the specified end point and configuration.
71       * @param endpoint The end point of the gitlab repository to access.
72       * @param configuration the configuration to use to access to gitlab.
73       * @throws ApplicationException in case the configuration is not valid.
74       */
75      Repository(final String endpoint, final ConnectionConfiguration configuration)
76          throws ApplicationException {
77          if (configuration.login() == null || configuration.login().isEmpty()
78              || configuration.password() == null || configuration.password().isEmpty()) {
79              throw new ApplicationException("The login and/or password cannot be empty");
80          }
81          this.configuration = configuration;
82          this.webb = Webb.create();
83          webb.setBaseUri(endpoint);
84      }
85  
86      /**
87       * Gives the comparator of version ids to use to be able to identify the latest version.
88       * @return the comparator of version ids.
89       */
90      Comparator<String> versionComparator() {
91          return this.configuration.versionComparator();
92      }
93  
94      /**
95       * Checks if the private token has already been retrieved.
96       * @return {@code true} if the private token has already been retrieved, {@code false} otherwise.
97       */
98      private boolean hasToken() {
99          return token != null;
100     }
101 
102     /**
103      * Retrieves the private token to use to acces to the gitlab repository.
104      * @return the private token to use to acces to the gitlab repository.
105      * @throws ApplicationException if the private token could not be retrieved.
106      */
107     private String findToken() throws ApplicationException {
108         if (hasToken()) {
109             return token;
110         }
111 
112         final Response<JSONObject> response;
113         try {
114             response = webb
115                  .post(String.format("/api/v3/session?login=%s&password=%s",
116                      URLEncoder.encode(configuration.login(), Repository.ENCODING),
117                      URLEncoder.encode(configuration.password(), Repository.ENCODING)))
118                  .asJsonObject();
119         } catch (UnsupportedEncodingException | WebbException e) {
120             throw new ApplicationException("Could not get the private token", e);
121         }
122 
123         final JSONObject body = response.getBody();
124         try {
125             if (response.getStatusCode() >= MIN_SUCCESS_CODE) {
126                 throw new ApplicationException(String.format("Could not connect to the server due to the error: %s",
127                     response.getResponseMessage()));
128             }
129             this.token = body.getString("private_token");
130         } catch (JSONException e) {
131             throw new ApplicationException("Could not extract the private token", e);
132         }
133         if (!hasToken()) {
134             throw new ApplicationException("No private token could be found");
135         }
136         return token;
137     }
138 
139     /**
140      * Gives the list of versions available in the repository ordered using version ids comparator.
141      * @return the of versions available.
142      * @throws ApplicationException if the list of versions could not be retrieved.
143      */
144     SortedSet<String> getVersions() throws ApplicationException {
145         final String token = findToken();
146         final Response<JSONArray> response;
147         try {
148             response = webb
149                 .get(String.format("/api/v3/projects/%s/repository/tree", configuration.projectId()))
150                 .param("private_token", token)
151                 .param("ref_name", configuration.branch())
152                 .asJsonArray();
153         } catch (WebbException e) {
154             throw new ApplicationException(String.format("Could not access to the versions of the project '%s",
155                 configuration.projectId()), e);
156         }
157         if (response.getStatusCode() >= MIN_SUCCESS_CODE) {
158             throw new ApplicationException(String.format(
159                 "Could not find the versions of the project '%s' in the branch '%s' due to the error: %s",
160                 configuration.projectId(),
161                 configuration.branch(),
162                 response.getResponseMessage()));
163         }
164         final JSONArray body = response.getBody();
165         final Comparator<String> comparator = configuration.versionComparator();
166         final SortedSet<String> result = new TreeSet<>(comparator);
167         for (int i = 0; i < body.length(); i++) {
168             try {
169                 final JSONObject jsonObject = body.getJSONObject(i);
170                 result.add(jsonObject.getString("name"));
171             } catch (JSONException e) {
172                 throw new ApplicationException("Could not extract the versions", e);
173             }
174         }
175         return result;
176     }
177 
178     /**
179      * Gets the content of the patch for the specified version id.
180      * @param version the version id for which we want the content of the patch.
181      * @return the content of the patch.
182      * @throws ApplicationException if the content of the patch could not be found.
183      */
184     InputStream getPatch(final String version) throws ApplicationException {
185         final String token = findToken();
186         final Response<InputStream> response;
187         try {
188             response = webb
189                 .get(String.format("/%s/%s/raw/%s/%s/%s",
190                     URLEncoder.encode(configuration.projectOwner(), Repository.ENCODING),
191                     URLEncoder.encode(configuration.projectName(), Repository.ENCODING),
192                     URLEncoder.encode(configuration.branch(), Repository.ENCODING),
193                     URLEncoder.encode(version, Repository.ENCODING),
194                     URLEncoder.encode(configuration.patchFileName(), Repository.ENCODING)))
195                 .param("private_token", token)
196                 .followRedirects(false)
197                 .asStream();
198         } catch (UnsupportedEncodingException | WebbException e) {
199             throw new ApplicationException(String.format("Could not access to the file '%s",
200                 configuration.patchFileName()), e);
201         }
202         if (response.getStatusCode() != OK_CODE) {
203             throw new ApplicationException(String.format(
204                 "Could not access to the file '%s due to the error: %s",
205                 configuration.patchFileName(), response.getResponseMessage()));
206         }
207         return response.getBody();
208     }
209 }
